[id="a-minimal-checluster-custom-resource-example"]
.A minimal `CheCluster` Custom Resource example.
====
[source,yaml,subs="+quotes,+attributes"]
----
apiVersion: org.eclipse.che/v2
kind: CheCluster
metadata:
  name: {prod-checluster}
spec:
  devEnvironments:
    defaultNamespace:
      template: '<username>-che'
    storage:
      pvcStrategy: 'common'
  components:
    metrics:
      enable: true
----
